   196→function rand(min: number, max: number): number {
   197→  return Math.floor(Math.random() * (max - min + 1)) + min;
   198→}
   199→
   200→function pick<T>(arr: T[]): T {
   201→  return arr[Math.floor(Math.random() * arr.length)];
   202→}
   203→
   204→function pickWeighted<T>(arr: T[], weights: number[]): T {
   205→  const total = weights.reduce((a, b) => a + b, 0);
   206→  let r = Math.random() * total;
   207→  for (let i = 0; i < arr.length; i++) {
   208→    r -= weights[i];
   209→    if (r <= 0) return arr[i];
   210→  }
   211→  return arr[arr.length - 1];
   212→}

   214→function slugify(str: string): string {
   215→  return str.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/^-|-$/g, "");
   216→}
